async function callGemini(prompt) {
  const res = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${GEMINI_KEY}`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }] })
  });
  const data = await res.json();
  return data.candidates?.[0]?.content?.parts?.[0]?.text || "";
}

function Auth({onAuth}) {
  const [v,setV]=useState("main");
  const [fb,setFb]=useState(null);
  const [phone,setPhone]=useState(""), [otp,setOtp]=useState(""), [confirm,setConfirm]=useState(null), [otpSent,setOtpSent]=useState(false);
  const [email,setEmail]=useState(""), [pass,setPass]=useState(""), [isNew,setIsNew]=useState(true);
  const [err,setErr]=useState(""), [loading,setLoading]=useState("");
  const [showP,setShowP]=useState(false);
  const recaptchaRef=useRef(), otpRef=useRef();

  // Load Firebase
  useEffect(()=>{
    const loadFirebase = async () => {
      try {
        // Load Firebase SDK
        if(!window.firebase) {
          await Promise.all([
            loadScript("https://www.gstatic.com/firebasejs/9.23.0/firebase-app-compat.js"),
            loadScript("https://www.gstatic.com/firebasejs/9.23.0/firebase-auth-compat.js"),
          ]);
        }
        if(!window.firebase.apps?.length) window.firebase.initializeApp(FB_CONFIG);
        setFb(window.firebase.auth());
      } catch(e) { console.error("Firebase load error",e); }
    };
    loadFirebase();
  },[]);

  const loadScript = (src) => new Promise((res,rej)=>{
    if(document.querySelector(`script[src="${src}"]`)){res();return;}
    const el=document.createElement("script"); el.src=src; el.onload=res; el.onerror=rej; document.head.appendChild(el);
  });

  const setupRecaptcha = () => {
    if(!fb) return null;
    if(window.recaptchaVerifier) { try{window.recaptchaVerifier.clear();}catch(e){} }
    window.recaptchaVerifier = new window.firebase.auth.RecaptchaVerifier(recaptchaRef.current, {
      size:"invisible", callback:()=>{}
    });
    return window.recaptchaVerifier;
  };

  const sendOtp = async () => {
    const num = "+91" + phone.replace(/\D/g,"");
    if(num.length < 13) { setErr("Enter valid 10-digit number"); return; }
    setErr(""); setLoading("otp");
    try {
      const rv = setupRecaptcha();
      const result = await fb.signInWithPhoneNumber(num, rv);
      setConfirm(result); setOtpSent(true); setLoading("");
      setTimeout(()=>otpRef.current?.focus(),100);
    } catch(e) {
      setLoading(""); setErr(e.message?.replace("Firebase: ","") || "Failed to send OTP");
      try{window.recaptchaVerifier?.clear();}catch(_){}
    }
  };

  const verifyOtp = async () => {
    if(!otp||otp.length<6) return;
    setLoading("vfy"); setErr("");
    try {
      const result = await confirm.confirm(otp);
      onAuth({ uid:result.user.uid, name:result.user.displayName||"User", phone:result.user.phoneNumber, method:"phone" });
    } catch(e) {
      setLoading(""); setErr("Invalid OTP. Please try again.");
    }
  };

  const emailAuth = async () => {
    if(!email.includes("@")) { setErr("Enter valid email"); return; }
    if(pass.length<8) { setErr("Min 8 characters required"); return; }
    if(!/[A-Z]/.test(pass)) { setErr("At least 1 uppercase letter required"); return; }
    if(!/[0-9]/.test(pass)) { setErr("At least 1 number required"); return; }
    setErr(""); setLoading("em");
    try {
      let r;
      if(isNew) r = await fb.createUserWithEmailAndPassword(email,pass);
      else      r = await fb.signInWithEmailAndPassword(email,pass);
      onAuth({ uid:r.user.uid, name:r.user.displayName||email.split("@")[0], email:r.user.email, method:"email" });
    } catch(e) {
      setLoading(""); setErr(e.message?.replace("Firebase: ","")?.replace(/\(.*\)/,"")?.trim() || "Authentication failed");
    }
  };

  const googleAuth = async () => {
    if(!fb) { setErr("Firebase not loaded yet, please wait"); return; }
    setLoading("g"); setErr("");
    try {
      const provider = new window.firebase.auth.GoogleAuthProvider();
      const r = await fb.signInWithPopup(provider);
      onAuth({ uid:r.user.uid, name:r.user.displayName||"User", email:r.user.email, photo:r.user.photoURL, method:"google" });
    } catch(e) {
      setLoading(""); setErr(e.message?.replace("Firebase: ","")?.replace(/\(.*\)/,"")?.trim() || "Google sign-in failed");
    }
  };

  // Phone view
  if(v==="phone") return (
    <div style={{...s.screen}} className="fi">
      <div ref={recaptchaRef} id="recaptcha-container" />
      <div style={{flex:1,padding:"52px 24px 40px",overflowY:"auto"}}>
        <button onClick={()=>{setV("main");setOtpSent(false);setOtp("");setPhone("");setErr("");}} style={{background:"none",border:"none",color:T.sub,cursor:"pointer",marginBottom:32,display:"flex",alignItems:"center",gap:6,fontSize:14,fontFamily:"'Plus Jakarta Sans',sans-serif"}}>
          <span style={{fontSize:18}}>←</span> Back
        </button>
        <div style={{fontSize:28,fontWeight:800,marginBottom:6,letterSpacing:-.5}}>{otpSent?"Verify OTP":"Phone Login"}</div>
        <div style={{color:T.sub,fontSize:14,marginBottom:28,lineHeight:1.5}}>
          {otpSent?`Code sent to +91 ${phone}\nFirebase SMS Auth`:"Secure login via Firebase OTP"}
        </div>
        {!otpSent?(
          <>
            <div style={{display:"flex",gap:10,marginBottom:12}}>
              <div style={{...s.input,flex:"none",width:"auto",padding:"15px 14px",color:T.sub,fontSize:14,display:"flex",alignItems:"center",gap:6,whiteSpace:"nowrap"}}>🇮🇳 +91</div>
              <input style={{...s.input,flex:1}} placeholder="10-digit mobile number" value={phone} onChange={e=>{setPhone(e.target.value.replace(/\D/g,"").slice(0,10));setErr("");}} type="tel" autoFocus />
            </div>
            {err&&<div style={{color:T.red,fontSize:12,marginBottom:12,paddingLeft:2,lineHeight:1.4}}>⚠ {err}</div>}
            <button style={{...s.btnP,opacity:phone.replace(/\D/g,"").length===10&&!loading?1:0.45}} onClick={sendOtp} disabled={!!loading}>
              {loading==="otp"?<Loader/>:"Send OTP via Firebase →"}
            </button>
            <div style={{marginTop:14,background:"rgba(0,229,160,0.06)",border:"1px solid rgba(0,229,160,0.15)",borderRadius:12,padding:"10px 14px",fontSize:12,color:T.mint,lineHeight:1.5}}>
              🔒 Real SMS via Firebase Auth — may take 10-30 seconds
            </div>
          </>
        ):(
          <>
            <div style={{position:"relative",marginBottom:10}}>
              <div style={{display:"flex",gap:8}}>
                {[0,1,2,3,4,5].map(i=>(
                  <div key={i} style={{flex:1,height:60,background:otp[i]?"rgba(0,255,255,0.1)":"rgba(44,44,46,.8)",border:`1.5px solid ${otp[i]?"#00ffff66":T.border}`,borderRadius:14,display:"flex",alignItems:"center",justifyContent:"center",fontSize:24,fontWeight:700,color:T.cyan,fontFamily:"'JetBrains Mono',monospace",transition:"all .2s"}}>
                    {otp[i]?"●":""}
                  </div>
                ))}
              </div>
              <input ref={otpRef} value={otp} onChange={e=>{setOtp(e.target.value.replace(/\D/g,"").slice(0,6));setErr("");}} style={{position:"absolute",opacity:0,inset:0,cursor:"pointer"}} type="tel" />
            </div>

function Quest({user, onComplete}) {
  const [step,setStep]=useState(0);
  const [d,setD]=useState({
    category:"", // school | competitive | college
    class:"", stream:"", subjects:[], examDates:{}, // per-subject dates
    compExam:"", year:"", degree:"",
    hours:4, struggles:[], name: user.name||""
  });
  const [k,setK]=useState(0);
  const [tip,setTip]=useState(false);

  const go = (patch={}) => { setD(x=>({...x,...patch})); setStep(s=>s+1); setK(x=>x+1); };
  const back = () => { setStep(s=>Math.max(0,s-1)); setK(x=>x+1); };

  // Determine current screen
  const sc = () => {
    if(step===0) return "name";
    if(step===1) return "category";
    if(d.category==="school") {
      if(step===2) return "class";
      if(step===3) return d.class==="Class 11"||d.class==="Class 12" ? "stream" : "subjects";
      if(step===4) return d.class==="Class 11"||d.class==="Class 12" ? "subjects" : "examDates";
      if(step===5) return d.class==="Class 11"||d.class==="Class 12" ? "examDates" : "hours";
      if(step===6) return d.class==="Class 11"||d.class==="Class 12" ? "hours" : "struggles";
      if(step===7) return "struggles";
      if(step===8) return "done";
    }
    if(d.category==="competitive") {
      if(step===2) return "compExam";
      if(step===3) return "year";
      if(step===4) return "examDates";
      if(step===5) return "hours";
      if(step===6) return "struggles";
      if(step===7) return "done";
    }
    if(d.category==="college") {
      if(step===2) return "degree";
      if(step===3) return "subjects";
      if(step===4) return "examDates";
      if(step===5) return "hours";
      if(step===6) return "struggles";
      if(step===7) return "done";
    }
    return "category";
  };
  const screen = sc();

  useEffect(()=>{ if(screen==="done") onComplete(d); },[screen]);

  const maxSt = d.category==="school"&&(d.class==="Class 11"||d.class==="Class 12") ? 8 :
                d.category==="school" ? 7 :
                d.category==="competitive" ? 7 : 7;
  const pct = Math.min(Math.round((step/maxSt)*100),98);

  // Available subjects
  const getAvailableSubjects = () => {
    if(d.category==="school") {
      if(d.class==="Class 11"||d.class==="Class 12") return SUBJECTS.school[d.class]?.streams?.[d.stream] || [];
      return SUBJECTS.school[d.class]?.subjects || [];
    }
    if(d.category==="competitive") return SUBJECTS.competitive[d.compExam]?.subjects || [];
    if(d.category==="college") return SUBJECTS.college[d.degree]?.subjects || [];
    return [];
  };

  const Radio = ({selected,color=T.cyan}) => (
    <div style={{width:20,height:20,borderRadius:10,border:`2px solid ${selected?color:"rgba(255,255,255,0.2)"}`,display:"flex",alignItems:"center",justifyContent:"center",flexShrink:0,transition:"all .2s"}}>
      {selected&&<div style={{width:8,height:8,borderRadius:4,background:color}} />}
    </div>
  );

  const struggles = [
    {id:"procrastinate",e:"😴",l:"I procrastinate a lot",n:"→ Strict daily task locks added"},
    {id:"forget",e:"🧠",l:"I forget what I study",n:"→ Spaced repetition schedule added"},
    {id:"syllabus",e:"😵",l:"Syllabus feels overwhelming",n:"→ Micro-goal daily breakdown"},
    {id:"distracted",e:"📱",l:"I get distracted easily",n:"→ Focus sessions recommended"},
  ];

  return (
    <div style={s.screen}>
      {/* Progress bar */}
      <div style={{padding:"48px 24px 0",flexShrink:0}}>
        <div style={{display:"flex",alignItems:"center",gap:12,marginBottom:4}}>
          {step>0&&<button onClick={back} style={{background:"none",border:"none",color:T.sub,fontSize:18,cursor:"pointer",padding:"4px 8px 4px 0"}}>←</button>}
          <div style={{flex:1}}><PBar pct={pct} /></div>
          <span style={{fontFamily:"'JetBrains Mono',monospace",fontSize:11,color:T.muted,flexShrink:0}}>{pct}%</span>
        </div>
      </div>

      <div key={k} className="fu" style={{flex:1,padding:"18px 24px 44px",overflowY:"auto"}}>

        {/* NAME */}
        {screen==="name"&&<>
          <div style={{fontSize:26,fontWeight:800,marginBottom:8,letterSpacing:-.5}}>What should your<br /><span className="gt">AI call you?</span></div>

// PROCESSING — Gemini generates real roadmap
// ════════════════════════════════════════════════════════════════
function Processing({user, qData, onDone}) {
  const [pct,setPct]=useState(0);
  const [msg,setMsg]=useState("Connecting to Gemini AI...");

  useEffect(()=>{
    generateRoadmap();
  },[]);

  const generateRoadmap = async () => {
    const steps = [
      [10,"Analysing your profile..."],
      [25,"Loading syllabus from AI..."],
      [45,"Building subject-wise schedule..."],
      [65,"Calculating daily study blocks..."],
      [80,"Optimising revision cycles..."],
      [95,"Finalising your roadmap..."],
    ];

    for(const [p,m] of steps) {
      setPct(p); setMsg(m);
      await new Promise(r=>setTimeout(r,600));
    }

    // Generate roadmap with Gemini
    const subs = qData.subjects.length ? qData.subjects : [];
    const daysLeft = Object.values(qData.examDates).length
      ? Math.min(...Object.values(qData.examDates).filter(Boolean).map(d=>Math.ceil((new Date(d)-new Date())/864e5)).filter(n=>n>0))
      : 90;

    const prompt = `You are a strict JSON API. Generate a study roadmap for a student.
Student: ${qData.name}, Category: ${qData.category}, Class/Exam: ${qData.class||qData.compExam||qData.degree}, Stream: ${qData.stream||qData.year||""}, Subjects: ${subs.join(", ")}, Hours/day: ${qData.hours}, Days left: ${daysLeft}, Struggles: ${qData.struggles.join(", ")}.
Return ONLY valid JSON (no markdown, no backticks):
{"subjects":[{"name":"","progress":0,"topics":[""],"color":"#hex","priority":"high|medium|low","weeklyHours":0}],"weekPlan":{"Mon":[],"Tue":[],"Wed":[],"Thu":[],"Fri":[],"Sat":[],"Sun":[]},"tips":[""],"focusSubject":"","totalDays":${daysLeft}}`;

    try {
      const raw = await callGemini(prompt);
      const cleaned = raw.replace(/```json|```/g,"").trim();
      const roadmap = JSON.parse(cleaned);
      setPct(100); setMsg("Roadmap ready ✓");
      await new Promise(r=>setTimeout(r,500));
      onDone(roadmap, daysLeft);
    } catch(e) {
      // Fallback roadmap
      const fallback = {
        subjects: subs.map((n,i)=>({name:n,progress:Math.floor(Math.random()*50+10),topics:["Chapter 1","Chapter 2","Chapter 3","Chapter 4","Chapter 5"],color:[T.cyan,T.mint,T.purple,T.orange,T.blue,T.red][i%6],priority:i<2?"high":"medium",weeklyHours:Math.floor(qData.hours/subs.length)||1})),
        weekPlan:{Mon:[subs[0]||""],Tue:[subs[1]||""],Wed:[subs[0]||""],Thu:[subs[2]||""],Fri:[subs[1]||""],Sat:subs.slice(0,3),Sun:["Revision"]},
        tips:["Start with your weakest subject every morning","Take 5-minute breaks every 25 minutes","Review yesterday's notes before starting new topic","Practice past papers in the last 2 weeks"],
        focusSubject:subs[0]||"",
        totalDays:daysLeft
      };
      setPct(100); setMsg("Roadmap ready ✓");
      await new Promise(r=>setTimeout(r,500));
      onDone(fallback, daysLeft);
    }
  };

  const R=100, C=2*Math.PI*R;
  return (
    <div style={{...s.screen,alignItems:"center",justifyContent:"center"}}>
      <div style={{textAlign:"center",padding:"0 32px"}}>
        <div style={{position:"relative",width:240,height:240,margin:"0 auto 36px"}}>
          <svg width="240" height="240" style={{position:"absolute",inset:0}}>
            <circle cx="120" cy="120" r={R} fill="none" stroke="rgba(255,255,255,0.05)" strokeWidth="7"/>
            <circle cx="120" cy="120" r={R} fill="none" stroke="url(#pg)" strokeWidth="7" strokeLinecap="round"
              strokeDasharray={C} strokeDashoffset={C*(1-pct/100)}
              style={{transform:"rotate(-90deg)",transformOrigin:"center",transition:"stroke-dashoffset .6s ease"}}
            />
            <defs><linearGradient id="pg" x1="0%" y1="0%" x2="100%" y2="0%"><stop offset="0%" stopColor="#00ffff"/><stop offset="100%" stopColor="#00e5a0"/></linearGradient></defs>
          </svg>
          <div style={{position:"absolute",inset:0,display:"flex",flexDirection:"column",alignItems:"center",justifyContent:"center"}}>
            <div style={{fontSize:38,fontWeight:800,color:T.cyan,lineHeight:1}}>{pct}%</div>
            <div style={{color:T.muted,fontSize:12,marginTop:4}}>Gemini AI</div>
          </div>
        </div>
        <div style={{fontSize:22,fontWeight:700,marginBottom:12,letterSpacing:-.5}}>Building Your AI OS</div>
        <div key={msg} className="fi" style={{color:T.cyan,fontSize:13,fontFamily:"'JetBrains Mono',monospace",minHeight:20,marginBottom:32}}>{msg}</div>
        <div style={{display:"flex",gap:6,justifyContent:"center"}}><span className="dot"/><span className="dot"/><span className="dot"/></div>
      </div>
    </div>
  );
}

// FOCUS — Real Timer + Real Audio
// ════════════════════════════════════════════════════════════════
function Focus() {
  const FULL=25*60;
  const [sec,setSec]=useState(FULL);
  const [run,setRun]=useState(false);
  const [strict,setStrict]=useState(false);
  const [music,setMusic]=useState(false);
  const [doubt,setDoubt]=useState(false);
  const [doubts,setDoubts]=useState([]);
  const [dTxt,setDTxt]=useState("");
  const [track,setTrack]=useState(0);
  const [session,setSession]=useState("Focus");
  const [sessions,setSessions]=useState(0);
  const [strictBlocked,setStrictBlocked]=useState(false);
  const audioRef=useRef(null);
  const timerRef=useRef();
  const gainRef=useRef(null);
  const audioCtxRef=useRef(null);

  // Real audio via Web Audio API
  const startAudio = (trackIdx) => {
    stopAudio();
    try {
      const ctx = new (window.AudioContext||window.webkitAudioContext)();
      audioCtxRef.current = ctx;
      const gain = ctx.createGain();
      gain.gain.value = 0.15;
      gain.connect(ctx.destination);
      gainRef.current = gain;

      const freqs = [[40,110,220],[432,528],[200,400,600],[261,329,392]]; // binaural-like
      const selectedFreqs = freqs[trackIdx%freqs.length];

      selectedFreqs.forEach((freq,i)=>{
        const osc = ctx.createOscillator();
        osc.type = ["sine","triangle","sawtooth","sine"][trackIdx%4];
        osc.frequency.setValueAtTime(freq, ctx.currentTime);
        if(trackIdx===1) { // binaural - slight detuning
          const osc2 = ctx.createOscillator();
          osc2.type = "sine";
          osc2.frequency.setValueAtTime(freq+10, ctx.currentTime);
          const g2 = ctx.createGain(); g2.gain.value=0.08; osc2.connect(g2); g2.connect(ctx.destination); osc2.start();
        }
        const g = ctx.createGain(); g.gain.value=0.06;
        osc.connect(g); g.connect(gain); osc.start();
        // Add tremolo for lo-fi feel
        if(trackIdx===0) {
          const lfo = ctx.createOscillator(); lfo.frequency.value=3;
          const lfoGain = ctx.createGain(); lfoGain.gain.value=0.04;
          lfo.connect(lfoGain); lfoGain.connect(g.gain); lfo.start();
        }
      });
    } catch(e) { console.log("Audio:",e); }
  };

  const stopAudio = () => {
    try { audioCtxRef.current?.close(); audioCtxRef.current=null; gainRef.current=null; } catch(e){}
  };

  useEffect(()=>{ if(music) startAudio(track); else stopAudio(); return stopAudio; },[music,track]);

  // Strict mode — intercepts navigation
  useEffect(()=>{
    if(strict&&run) {
      const handler = (e) => { e.preventDefault(); setStrictBlocked(true); setTimeout(()=>setStrictBlocked(false),2000); return false; };
      window.addEventListener("beforeunload",handler);
      return ()=>window.removeEventListener("beforeunload",handler);
    }
  },[strict,run]);

  useEffect(()=>{
    if(run&&sec>0) timerRef.current=setInterval(()=>setSec(s=>s-1),1000);
    else { clearInterval(timerRef.current); if(sec===0&&run){setSessions(s=>s+1);setRun(false);setSec(5*60);setSession("Break");if("Notification" in window&&Notification.permission==="granted") new Notification("⏰ Qyro",{body:"Focus session complete! Take a break 🎉"});}}
    return()=>clearInterval(timerRef.current);
  },[run,sec]);

  const mm=String(Math.floor(sec/60)).padStart(2,"0"), ss=String(sec%60).padStart(2,"0");
  const R=100, C=2*Math.PI*R, off=C*(sec/FULL);
  const tracks=[{l:"Lo-Fi Study",e:"🎵",desc:"Soft ambient beats"},{l:"Binaural Focus",e:"🧠",desc:"432Hz Alpha waves"},{l:"Rain & Ambient",e:"🌧",desc:"Brown noise + rain"},{l:"Classical",e:"🎹",desc:"Piano focus music"}];

  return (
    <div style={{...s.screen,paddingTop:0}}>
      {strictBlocked&&(
        <div className="fi" style={{position:"fixed",inset:0,background:"rgba(255,69,58,.15)",border:"2px solid rgba(255,69,58,.4)",zIndex:300,display:"flex",alignItems:"center",justifyContent:"center",backdropFilter:"blur(20px)"}}>
          <div style={{textAlign:"center",padding:32}}>
            <div style={{fontSize:48,marginBottom:12}}>🔒</div>
            <div style={{fontSize:22,fontWeight:700,color:T.red,marginBottom:8}}>Strict Mode Active</div>
            <div style={{color:T.sub,fontSize:14}}>Stay focused! Complete your session first.</div>
          </div>
        </div>
      )}
      <div style={{flex:1,overflowY:"auto",paddingTop:52,paddingBottom:84}}>
        <div style={{padding:"0 20px 14px"}}>
          <div style={{fontSize:11,color:T.muted,fontFamily:"'JetBrains Mono',monospace",marginBottom:4}}>FOCUS ROOM</div>
          <div style={{fontSize:22,fontWeight:700,letterSpacing:-.5}}>Pomodoro Timer</div>
        </div>

        {/* Session pills */}
        <div style={{padding:"0 20px",display:"flex",gap:8,marginBottom:20}}>
          {[["Focus",25*60],["Short Break",5*60],["Long Break",15*60]].map(([sn,t])=>(
            <div key={sn} onClick={()=>{setSession(sn);setRun(false);setSec(t);}} style={{background:session===sn?"rgba(0,255,255,.1)":"transparent",border:`0.5px solid ${session===sn?"rgba(0,255,255,.3)":T.border}`,borderRadius:10,padding:"6px 13px",fontSize:11,color:session===sn?T.cyan:T.muted,cursor:"pointer",whiteSpace:"nowrap",transition:"all .2s"}}>
              {sn}
            </div>
          ))}
        </div>

        {/* Timer */}
        <div style={{display:"flex",flexDirection:"column",alignItems:"center",paddingBottom:24}}>
          <div style={{position:"relative",width:240,height:240}}>
            <svg width="240" height="240" style={{position:"absolute",inset:0}}>
              <circle cx="120" cy="120" r={R} fill="none" stroke="rgba(255,255,255,0.04)" strokeWidth="7"/>
              <circle cx="120" cy="120" r={R} fill="none" stroke="url(#tg)" strokeWidth="7" strokeLinecap="round"
                strokeDasharray={C} strokeDashoffset={C-off}
                style={{transform:"rotate(-90deg)",transformOrigin:"center",transition:"stroke-dashoffset 1s linear",filter:run?"drop-shadow(0 0 10px rgba(0,255,255,.6))":"none"}}

              <defs><linearGradient id="tg" x1="0%" y1="0%" x2="100%" y2="100%"><stop offset="0%" stopColor="#00ffff"/><stop offset="100%" stopColor="#00e5a0"/></linearGradient></defs>
            </svg>
            <div style={{position:"absolute",inset:0,display:"flex",flexDirection:"column",alignItems:"center",justifyContent:"center"}}>
              <div style={{fontFamily:"'JetBrains Mono',monospace",fontSize:52,fontWeight:300,letterSpacing:-3,lineHeight:1,animation:run?"timerP 2s ease-in-out infinite":"none"}}>{mm}:{ss}</div>
              <div style={{color:T.muted,fontSize:12,marginTop:8}}>{session} • #{sessions+1}</div>
            </div>
          </div>
          <div style={{display:"flex",gap:16,alignItems:"center",marginTop:14}}>
            <button onClick={()=>{setRun(false);setSec(FULL);}} style={{width:46,height:46,background:"rgba(44,44,46,.8)",border:`0.5px solid ${T.border}`,borderRadius:14,color:T.muted,fontSize:18,cursor:"pointer"}}>↺</button>

        </div>

        {/* Strict mode */}
        <div style={{margin:"0 20px 10px"}}>
          <div style={{...s.card,display:"flex",alignItems:"center",justifyContent:"space-between"}}>
            <div>

  const [roadmap,setRoadmap]=useState(null);

  const [theme,setTheme]=useState("cyan");

  const themeAccent = {
    cyan:{a:"#00ffff",b:"#00e5a0"},
    purple:{a:"#bf5af2",b:"#6e6ef5"},
    orange:{a:"#ff9f0a",b:"#ff453a"},
    blue:{a:"#0a84ff",b:"#00e5a0"},
    gold:{a:"#ffd60a",b:"#ff9f0a"},
  }[theme]||{a:"#00ffff",b:"#00e5a0"};

  const themeCSS = `
    .gt { background: linear-gradient(135deg,${themeAccent.a},${themeAccent.b}); -webkit-background-clip:text; -webkit-text-fill-color:transparent; background-clip:text; }
    .dot { background: ${themeAccent.a}; }
  `;

    {id:"home",icon:"⊞",label:"Home"},
    {id:"roadmap",icon:"⊟",label:"Roadmap"},
    {id:"focus",icon:"◎",label:"Focus"},
    {id:"profile",icon:"◉",label:"Profile"},
  ];

  return (
    <>
      <style dangerouslySetInnerHTML={{__html:CSS+themeCSS}} />
      <div style={{position:"fixed",inset:0,background:T.bg}}>
        {phase==="splash"&&<Splash onDone={()=>setPhase("auth")} />}
        {phase==="auth"&&<Auth onAuth={u=>{setUser(u);setPhase("quest");}} />}
        {phase==="quest"&&user&&<Quest user={user} onComplete={qd=>{setQData(qd);setPhase("processing");}} />}

          <>
